// CHANGED: The main interface for all problems in the application.

// The 'answer' type is set to string for consistency across all data files.

  category: string;
  difficulty: number;
  problem: string;
  answer: string; // Changed to just string for consistency
  unit?: string; // ADDED: Optional field for the answer's unit

}
